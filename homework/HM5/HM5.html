PCA and the standard autoencoder are unsupervised dimensionality reduction methods, and their learned features are not discriminative. If you build a classifier upon the low-dimenional features extracted by PCA and autoencoder, you will find the classification accuracy very poor.

Linear discriminant analysis (LDA) is a traditionally supervised dimensionality reduction method for learning low-dimensional features which are highly discriminative. Likewise, can we extend autoencoder to supervised leanring?

import numpy

def to_one_hot(y, num_class=10):
    results = numpy.zeros((len(y), num_class))
    for i, label in enumerate(y):
        results[i, label] = 1.
    return results

# build the supervised autoencoder network
from keras.layers import Dense, Input, Activation, BatchNormalization, LeakyReLU, Dropout
from keras import models
from keras import regularizers

input_img = Input(shape=(784,), name='input_img')
encode1 = Dropout(0.5)(input_img)

encode1 = Dense(256, name='encode1')(encode1)
encode1 = BatchNormalization(name='bn1')(encode1)
encode1 = LeakyReLU(name='ac1')(encode1)

encode2 = Dropout(0.5)(encode1)
encode2 = Dense(256, name='encode2')(encode2)
encode2 = BatchNormalization(name='bn2')(encode2)
encode2 = LeakyReLU(name='ac2')(encode2)

encode3 = Dense(128, name='encode3')(encode2)
encode3 = BatchNormalization(name='bn3')(encode3)
encode3 = LeakyReLU(name='ac3')(encode3)

encode4 = Dense(32, name='encode4')(encode3)
encode4 = BatchNormalization(name='bn4')(encode4)
encode4 = LeakyReLU(name='ac4')(encode4)

bottleneck = Dense(2, name='bottleneck')(encode4)
bottleneck = BatchNormalization(name='bn5')(bottleneck)
bottleneck = LeakyReLU(name='ac5')(bottleneck)

decode1 = Dense(32, name='decode1')(bottleneck)
decode1 = BatchNormalization(name='bn6')(decode1)
decode1 = LeakyReLU(name='ac6')(decode1)

decode2 = Dense(128, name='decode2')(decode1)
decode2 = BatchNormalization(name='bn7')(decode2)
decode2 = LeakyReLU(name='ac7')(decode2)

decode3 = Dense(256, name='decode3')(decode2)
decode3 = BatchNormalization(name='bn8')(decode3)
decode3 = LeakyReLU(name='ac8')(decode3)

decode4 = Dropout(0.5)(decode3)
decode4 = Dense(256, name='decode4')(decode4)
decode4 = BatchNormalization(name='bn9')(decode4)
decode4 = LeakyReLU(name='ac9')(decode4)

decode5 = Dense(784, activation='relu', name='decode5')(decode4)

# build a classifier upon the bottleneck layer
classifier1 = Dense(128)(bottleneck)
classifier1 = BatchNormalization(name='bn10')(classifier1)
classifier1 = LeakyReLU(name='ac10')(classifier1)
classifier2 = Dense(128)(classifier1)
classifier2 = BatchNormalization(name='bn11')(classifier2)
classifier2 = LeakyReLU(name='ac11')(classifier2)
classifier3 = Dense(64)(classifier2)
classifier3 = BatchNormalization(name='bn12')(classifier3)
classifier3 = LeakyReLU(name='ac12')(classifier3)
classifier4 = Dense(10, activation='softmax')(classifier3)
